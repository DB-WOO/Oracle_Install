● 9. .bash_profile 파일 구성<br>
<br>
--------- Oracle 계정 --------- <br>
<br>
# .bash_profile                                                                                             <br>
if [ -f ~/.bashrc ]; then                                                                                   <br>
    . ~/.bashrc                                                                                             <br>
fi                                                                                                          <br>
																											<br>
																											<br>
PATH=$PATH:$HOME/.local/bin:$HOME/bin                                                                       <br>
																											<br>
export PATH                                                                                                 <br>
export TMP=/tmp                                                                                             <br>
export TMPDIR=$TMP                                                                                          <br>
																											<br>
export ORACLE_HOSTNAME=asmtest                                                                              <br>
export ORACLE_UNQNAME=orcl                                                                                  <br>
export ORACLE_BASE=/app/oracle                                                                              <br>
export ORACLE_HOME=$ORACLE_BASE/product/19.0.0/db_1                                                         <br>
export GRID_HOME=/app/grid                                                                                  <br>
export ORA_INVENTORY=/oralnventory                                                                          <br>
export ORACLE_SID=orcl                                                                                      <br>
																											<br>
# 공백 제거 + ORACLE_HOME/bin 은 PATH 앞에 추가 (권장)                                                        <br>
export PATH=/usr/sbin:/usr/local/bin:$PATH                                                                  <br>
export PATH=$ORACLE_HOME/bin:$PATH                                                                          <br>
																											<br>
export LD_LIBRARY_PATH=$ORACLE_HOME/lib:/lib:/usr/lib                                                       <br>
export CLASSPATH=$ORACLE_HOME/jlib:$ORACLE_HOME/rdbms/jlib                                                  <br>
																											<br>
export CV_ASSUME_DISTID=RHEL7.6                                                                             <br>
																											<br>
export LANG=C                                                                                               <br>
export LC_ALL=C                                                                                             <br>
																											<br>
alias grid_env='export ORACLE_HOME=$GRID_HOME; export ORACLE_SID=+ASM; export PATH=$GRID_HOME/bin:$PATH'    <br>
alias db_env='export ORACLE_HOME=$ORACLE_BASE/product/19.0.0/db_1; export ORACLE_SID=orcl'                  <br>
alias ss='sqlplus / as sysdba'                                                                              <br>
alias oh='cd $ORACLE_HOME'                                                                                  <br>
alias gh='cd $GRID_HOME'                                                                                    <br>
<br>
<br>
<br>
<br>

<br>
# .bash_profile<br>
<br>
# Get the aliases and functions<br>

# User specific environment and startup programs<br>
<br>

<br>
